# lets make sure the allies dont overcommit to invasions early on
careful_with_early_invasions = {
	enable = {
		is_historical_focus_on = yes
		ITA = { is_ai = yes }

		is_in_faction_with = ENG # anyone in allies

		date < 1942.1.1
		NOT = {
			GER = { has_war_with = SOV }
		}
	}
	abort = {
		OR = {
			GER = { has_war_with = SOV }
			date > 1942.1.1
		}		
	}

	ai_strategy = {
		type = invade
		id = "ITA"			
		value = -100
	}

	ai_strategy = {
		type = invade
		id = "GER"			
		value = -80
	}
}
